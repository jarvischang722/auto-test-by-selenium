# 1. npm install 

# 2. 安裝 codeceptjs
    $ npm install -g codeceptjs-webdriverio  
    or 
    $ npm install codeceptjs-webdriverio --save-dev
    
# 3. 安裝 selenium-standalone 
    $ npm install selenium-standalone@latest -g
    $ selenium-standalone install
    
    打開終端機執行
    $ selenium-standalone start  
    
# 4. 開始執行測試程序
    npm run auto-test
